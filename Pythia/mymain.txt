double deltaPHI(double phi1,double phi2){
	double R;
	R=phi1-phi2;
	if (R>M_PI){R=2*M_PI-R;
	}
	if (R<-M_PI){R=R+2*M_PI;
	}
	return R;
} 
double deltaR(double phi1,double phi2,double eta1,double eta2){
	double R;
	R=sqrt(pow(deltaPHI(phi1,phi2),2)+pow(eta1-eta2,2));
	return R;
}
